🚀 HOW TO USE
═══════════════════════════════════════════════════════════════════════════════

Step 1: In Program.cs
───────────────────────
using Application.Mapping;

var app = builder.Build();

// Initialize the Mapper logger
var logger = app.Services.GetRequiredService<ILogger<Program>>();
var mapperLogger = LoggerFactory.Create(config => config.AddConsole())
    .CreateLogger("Application.Mapping.Mapper");
Mapper.SetLogger(mapperLogger);

Step 2: Use Mapper Normally
──────────────────────────
var products = await _productService.GetAllAsync();
var productDtos = Mapper.MapToDtoList(products, languageId);
// ✓ All logging happens automatically!

⚙️ CONFIGURATION OPTIONS
═══════════════════════════════════════════════════════════════════════════════

Option 1: Console Logging
─────────────────────────
builder.Services.AddLogging(config =>
{
    config.AddConsole();
});

Option 2: File Logging (with Serilog)
──────────────────────────────────────
builder.Services.AddLogging(config =>
{
    config.AddFile("logs/mapper-{Date}.txt");
});

Option 3: Conditional Logging by Level
───────────────────────────────────────
builder.Services.AddLogging(config =>
{
    config.AddConsole();
    config.SetMinimumLevel(LogLevel.Warning); // Only warnings and errors
});

Option 4: Logger Factory Approach
──────────────────────────────────
var loggerFactory = LoggerFactory.Create(config =>
{
    config.AddConsole();
    config.AddDebug();
});
var logger = loggerFactory.CreateLogger("Mapper");
Mapper.SetLogger(logger);
